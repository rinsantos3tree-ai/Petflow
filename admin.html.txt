<script>
    const firebaseConfig = { apiKey: "AIzaSyAq4NH16AM80LjdfDSD53T1XtsurbCn98A", authDomain: "petflow-ia.firebaseapp.com", projectId: "petflow-ia" };
    firebase.initializeApp(firebaseConfig);
    const auth = firebase.auth(); const db = firebase.firestore();
    
    let idParaExcluir = null;
    let modoCad = false;
    let chartObj;

    function avisar(m, c="#10b981") { 
        const t = document.getElementById('toast'); t.innerText=m; t.style.background=c; t.style.display='block'; 
        setTimeout(()=>t.style.display='none', 3000); 
    }

    function configurarBoasVindas() {
        const h = new Date().getHours();
        const frases = [
            "O sucesso é a soma de pequenos esforços diários.",
            "Transforme cuidado em uma experiência de luxo.",
            "Foque na excelência e o faturamento virá.",
            "Organização é a chave para escalar seu negócio.",
            "Mais um dia para elevar o nível da sua estética pet!"
        ];
        document.getElementById('saudacao').innerText = h < 12 ? "Bom dia! ☀️" : h < 18 ? "Boa tarde! 🔥" : "Boa noite! 🌙";
        document.getElementById('fraseImpacto').innerText = frases[Math.floor(Math.random() * frases.length)];
    }

    async function esqueciSenha() {
        const e = document.getElementById('uEmail').value;
        if(!e) return avisar("Digite seu e-mail acima primeiro", "#ef4444");
        auth.sendPasswordResetEmail(e).then(() => avisar("E-mail enviado!")).catch(() => avisar("Erro","#ef4444"));
    }

    function toggleModo() { modoCad = !modoCad; document.getElementById('btnToggle').innerText = modoCad ? "Já tenho conta" : "Criar empresa grátis"; }

    async function gerenciarAcesso() {
        const e=document.getElementById('uEmail').value, s=document.getElementById('uSenha').value;
        try { if(modoCad) await auth.createUserWithEmailAndPassword(e,s); else await auth.signInWithEmailAndPassword(e,s); } 
        catch(err) { avisar("Erro no acesso","#ef4444"); }
    }

    function abrirModal(id) { idParaExcluir = id; document.getElementById('modalExcluir').style.display = 'flex'; }
    function fecharModal() { document.getElementById('modalExcluir').style.display = 'none'; }

    document.getElementById('confirmarBtn').onclick = async () => {
        if(idParaExcluir) { await db.collection("agendamentos").doc(idParaExcluir).delete(); fecharModal(); avisar("Removido!", "#ef4444"); }
    };

    function filtrar() {
        const termo = document.getElementById('campoBusca').value.toLowerCase();
        document.querySelectorAll('.pet-item').forEach(i => { i.style.display = i.dataset.nome.toLowerCase().includes(termo) ? 'block' : 'none'; });
    }

    async function addPet() {
        const inputH = document.getElementById('pHorario');
        const n=document.getElementById('pNome').value, h=inputH.value;
        if(!n || !h) return avisar("Preencha Nome e Hora","#ef4444");
        await db.collection("agendamentos").add({
            uid: auth.currentUser.uid, nome: n, raca: document.getElementById('pRaca').value || "SRD",
            servico: document.getElementById('pServico').value || "Serviço Geral", horario: h,
            valor: parseFloat(document.getElementById('pValor').value) || 0,
            whats: document.getElementById('pWhats').value.replace(/\D/g,''),
            obs: document.getElementById('pObs').value || "", status: 'st-agendado', txt: 'Agendado', criado: Date.now(), dataSimples: new Date().toLocaleDateString()
        });
        avisar("Pet Agendado!");
        document.querySelectorAll('input, select, textarea').forEach(i => i.value = '');
        inputH.type = 'text';
    }

    function carregar(uid) {
        db.collection("agendamentos").where("uid", "==", uid).onSnapshot(snap => {
            let html="", fatDia=0, petsDia=0, hist={}, arr=[];
            const hoje = new Date().toLocaleDateString();
            snap.forEach(doc => arr.push({id: doc.id, ...doc.data()}));
            arr.sort((a,b) => (a.horario||"").localeCompare(b.horario||""));
            arr.forEach(d => {
                if(d.status === 'st-pronto' && d.dataSimples === hoje) { fatDia += d.valor; petsDia++; }
                if(d.status === 'st-pronto') hist[d.dataSimples] = (hist[d.dataSimples] || 0) + d.valor;
                const link = `https://petflow.vercel.app/status.html?id=${d.id}`;



                const msg = encodeURIComponent(`Olá! Acompanhe o status do serviço do ${d.nome} aqui: ${link}`);
                
                // BLOCO DE OBSERVAÇÃO ADICIONADO ABAIXO
                let obsHtml = d.obs ? `<p style="color: #94a3b8; font-size: 11px; margin-bottom: 12px; background: rgba(0,0,0,0.2); padding: 8px; border-radius: 10px;">📝 ${d.obs}</p>` : "";

                html += `<div class="pet-card pet-item" data-nome="${d.nome}">
                    <button style="position:absolute; right:20px; color:#ef4444; border:none; background:none; font-weight:800; font-size:22px;" onclick="abrirModal('${d.id}')">✕</button>
                    <small style="color:var(--p); font-weight:800;">⏰ ${d.horario}</small>
                    <h3 style="margin:5px 0;">${d.nome} <small style="color:gray; font-size:12px">(${d.raca})</small></h3>
                    <p style="color: #64748b; font-size: 11px; font-weight: 700; margin-bottom: 8px;">🛠 ${d.servico}</p>
                    ${obsHtml}
                    <button class="status-badge ${d.status}" onclick="mudarStatus('${d.id}','${d.status}')">${d.txt}</button>
                    <div style="display:grid; grid-template-columns: 1fr 1fr; gap:10px;">
                        <a href="https://wa.me/55${d.whats}" target="_blank" class="btn-p" style="background:#10b981; padding:15px; box-shadow:none;">WHATSAPP</a>
                        <a href="https://wa.me/55${d.whats}?text=${msg}" target="_blank" class="btn-p" style="background:#3b82f6; padding:15px; box-shadow:none;">STATUS ONLINE</a>
                    </div>
                </div>`;
            });
            document.getElementById('valFaturado').innerText = `R$ ${fatDia.toFixed(2)}`;
            document.getElementById('valConcluidos').innerText = petsDia;
            document.getElementById('lista').innerHTML = html || "<p style='text-align:center; color:gray; padding:20px;'>Lista vazia.</p>";
            renderGrafico(hist);
        });
    }

    function mudarStatus(id, s) {
        const ciclo = { 'st-agendado':['st-servico','Em Serviço'], 'st-servico':['st-pronto','Pronto!'], 'st-pronto':['st-agendado','Agendado'] };
        db.collection("agendamentos").doc(id).update({ status: ciclo[s][0], txt: ciclo[s][1] });
    }

    function renderGrafico(h) {
        const ctx = document.getElementById('graficoVendas').getContext('2d');
        const labels = Object.keys(h).sort().slice(-7);
        if(chartObj) chartObj.destroy();
        chartObj = new Chart(ctx, { type: 'line', data: { labels, datasets: [{ data: labels.map(l=>h[l]), borderColor: '#f97316', borderWidth: 3, tension: 0.4, fill: true, backgroundColor: 'rgba(249, 115, 22, 0.1)' }] }, options: { maintainAspectRatio: false, plugins: { legend: { display: false } }, scales: { y: { display: false }, x: { grid: { display: false }, ticks: { color: '#64748b', font: {size: 10, weight: '700'} } } } } });
    }

    auth.onAuthStateChanged(u => {
        document.querySelectorAll('.tela').forEach(t => t.classList.remove('ativa'));
        if(u) { document.getElementById('telaDash').classList.add('ativa'); configurarBoasVindas(); carregar(u.uid); } 
        else { document.getElementById('telaAcesso').classList.add('ativa'); }
    });
